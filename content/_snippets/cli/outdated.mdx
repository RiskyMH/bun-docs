## CLI Usage

```bash title="terminal"
$ bun outdated <filter>
```

### General Options

<ParamField name="-c, --config" type="string">
	Specify path to config file (<code>bunfig.toml</code>)
</ParamField>

<ParamField name="--cwd" type="string">
	Set a specific cwd
</ParamField>

<ParamField name="-h, --help" type="boolean">
	Print this help menu
</ParamField>

<ParamField name="-F, --filter" type="string">
	Display outdated dependencies for each matching workspace
</ParamField>

### Output & Logging

<ParamField name="--silent" type="boolean">
	Don't log anything
</ParamField>

<ParamField name="--verbose" type="boolean">
	Excessively verbose logging
</ParamField>

<ParamField name="--no-progress" type="boolean">
	Disable the progress bar
</ParamField>

<ParamField name="--no-summary" type="boolean">
	Don't print a summary
</ParamField>

### Dependency Scope & Target

<ParamField name="-p, --production" type="boolean">
	Don't install devDependencies
</ParamField>

<ParamField name="--omit" type="string">
	Exclude <code>dev</code>, <code>optional</code>, or <code>peer</code> dependencies from install
</ParamField>

<ParamField name="-g, --global" type="boolean">
	Install globally
</ParamField>

### Lockfile & Package.json

<ParamField name="-y, --yarn" type="boolean">
	Write a <code>yarn.lock</code> file (yarn v1)
</ParamField>

<ParamField name="--no-save" type="boolean">
	Don't update <code>package.json</code> or save a lockfile
</ParamField>

<ParamField name="--save" type="boolean" default="true">
	Save to <code>package.json</code> (true by default)
</ParamField>

<ParamField name="--frozen-lockfile" type="boolean">
	Disallow changes to lockfile
</ParamField>

<ParamField name="--save-text-lockfile" type="boolean">
	Save a text-based lockfile
</ParamField>

<ParamField name="--lockfile-only" type="boolean">
	Generate a lockfile without installing dependencies
</ParamField>

<ParamField name="--trust" type="boolean">
	Add to <code>trustedDependencies</code> in the project's <code>package.json</code> and install the
	package(s)
</ParamField>

### Network & Registry

<ParamField name="--ca" type="string">
	Provide a Certificate Authority signing certificate
</ParamField>

<ParamField name="--cafile" type="string">
	Same as <code>--ca</code>, but as a file path to the certificate
</ParamField>

<ParamField name="--registry" type="string">
	Use a specific registry by default, overriding <code>.npmrc</code>, <code>bunfig.toml</code> and
	environment variables
</ParamField>

<ParamField name="--network-concurrency" type="number" default="48">
	Maximum number of concurrent network requests (default 48)
</ParamField>

### Caching

<ParamField name="--cache-dir" type="string">
	Store &amp; load cached data from a specific directory path
</ParamField>

<ParamField name="--no-cache" type="boolean">
	Ignore manifest cache entirely
</ParamField>

### Execution Behavior

<ParamField name="--dry-run" type="boolean">
	Don't install anything
</ParamField>

<ParamField name="-f, --force" type="boolean">
	Always request the latest versions from the registry &amp; reinstall all dependencies
</ParamField>

<ParamField name="--no-verify" type="boolean">
	Skip verifying integrity of newly downloaded packages
</ParamField>

<ParamField name="--ignore-scripts" type="boolean">
	Skip lifecycle scripts in the project's <code>package.json</code> (dependency scripts are never
	run)
</ParamField>

<ParamField name="--backend" type="string" default="clonefile">
	Platform-specific optimizations for installing dependencies. Possible values:{' '}
	<code>clonefile</code> (default), <code>hardlink</code>, <code>symlink</code>,{' '}
	<code>copyfile</code>
</ParamField>

<ParamField name="--concurrent-scripts" type="number" default="5">
	Maximum number of concurrent jobs for lifecycle scripts (default 5)
</ParamField>
