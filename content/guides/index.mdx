import { readdirSync, readFileSync } from 'node:fs';
import { join } from 'node:path';
import { Card, Cards } from 'fumadocs-ui/components/card';

export const GuidesCategory = ({ category }) => {
  const guidesDir = join(process.cwd(), 'content/guides', category);
  const files = readdirSync(guidesDir).filter(f => f.endsWith('.mdx'));
  
  const pages = files.map(file => {
    const content = readFileSync(join(guidesDir, file), 'utf-8');
    
    const frontmatterMatch = content.match(/^---\n([\s\S]*?)\n---/);
    const frontmatter = frontmatterMatch ? frontmatterMatch[1] : '';
    
    const titleMatch = frontmatter.match(/^title:\s*['"]?(.+?)['"]?\s*$/m);
    const fullTitleMatch = frontmatter.match(/^fullTitle:\s*['"]?(.+?)['"]?\s*$/m);
    
    const pageTitle = titleMatch ? titleMatch[1] : file.replace('.mdx', '');
    const pageFullTitle = fullTitleMatch ? fullTitleMatch[1] : pageTitle;
    
    return {
      title: pageFullTitle ||pageTitle,
      url: `/guides/${category}/${file.replace('.mdx', '')}`
    };
  }).sort((a, b) => a.title.localeCompare(b.title));

  return (
    <Cards className="grid-cols-3">
      {pages.map((page) => (
        <Card 
          key={page.url} 
          title={page.title} 
          href={page.url}
        />
      ))}
    </Cards>
  );
};
